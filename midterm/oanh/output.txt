CREATE TABLE ChiTietHoaDon
(
	PRIMARY KEY(MaSP)
	PRIMARY KEY(MaHD)
	FOREIGN KEY(MaSP) REFERENCES Sanpham(MaSP)
	FOREIGN KEY(MaHD) REFERENCES Hoadon(MaHD)
)
CREATE TABLE Daily(
	PRIMARY KEY( MaDL )
	ATTRIBUTE TenchuDL
	ATTRIBUTE Diachi
	ATTRIBUTE SDT
	ATTRIBUTE Thanhtoan
)
CREATE TABLE Hoadon(
	PRIMARY KEY( MaHD )
	ATTRIBUTE Ngaylap
	FOREIGN KEY(MaDL) REFERENCES Daily(MaDL)
)
CREATE TABLE Nhanvien(
	PRIMARY KEY(MaTN)
	PRIMARY KEY( MaNV )
	ATTRIBUTE Hovaten
	ATTRIBUTE SDT
	ATTRIBUTE Diachi
	FOREIGN KEY(MaDL) REFERENCES Daily(MaDL)
	FOREIGN KEY(MaTN) REFERENCES Thannhan(MaTN)
)
CREATE TABLE Thannhan
(
	PRIMARY KEY( MaTN )
	ATTRIBUTE Hovaten
)
CREATE TABLE Sanpham(
	PRIMARY KEY( MaSP )
	ATTRIBUTE TenSP
	ATTRIBUTE Giaban
	ATTRIBUTE Soluongtonkho
)
CREATE TABLE Quanao(
	FOREIGN KEY(MaSP)
	ATTRIBUTE Chatvai
	ATTRIBUTE Nhasanxuat
	ATTRIBUTE Noisanxuat
	FOREIGN KEY(MaSP) REFERENCES Sanpham(MaSP)
)
CREATE TABLE Soi(
	FOREIGN KEY(MaSP)
	ATTRIBUTE Chatvai
	ATTRIBUTE Nhasanxuat
	ATTRIBUTE Noisanxuat
	FOREIGN KEY(MaSP) REFERENCES Sanpham(MaSP)
)
